   185‚Üí   185‚Üí  async runMigrations(): Promise<void> {
   186‚Üí   186‚Üí    for (const migration of MIGRATIONS) {
   187‚Üí   187‚Üí      await this.pool.query('BEGIN');
   188‚Üí   188‚Üí      try {
   189‚Üí   189‚Üí        // Check if already applied
   190‚Üí   190‚Üí        const check = await this.pool.query(
   191‚Üí   191‚Üí          'SELECT version FROM si_platform.migrations WHERE version = $1',
   192‚Üí   192‚Üí          [migration.version],
   193‚Üí   193‚Üí        );
   194‚Üí   194‚Üí        if (check.rows.length > 0) continue;
   195‚Üí   195‚Üí
   196‚Üí   196‚Üí        // Apply migration
   197‚Üí   197‚Üí        await this.pool.query(migration.up);
   198‚Üí   198‚Üí
   199‚Üí   199‚Üí        // Record migration
   200‚Üí   200‚Üí        const checksum = this.simpleChecksum(migration.up);
   201‚Üí   201‚Üí        await this.pool.query(
   202‚Üí   202‚Üí          'INSERT INTO si_platform.migrations (version, description, checksum, applied_at) VALUES ($1, $2, $3, NOW())',
   203‚Üí   203‚Üí          [migration.version, migration.description, checksum],
   204‚Üí   204‚Üí        );
   205‚Üí   205‚Üí
   206‚Üí   206‚Üí        await this.pool.query('COMMIT');
   207‚Üí   207‚Üí      } catch (err) {
   208‚Üí   208‚Üí        await this.pool.query('ROLLBACK');
   209‚Üí   209‚Üí        throw new Error(`Migration ${migration.version} failed: ${err instanceof Error ? err.message : String(err)}`);
   210‚Üí   210‚Üí      }
   211‚Üí   211‚Üí    }
   212‚Üí   212‚Üí  }

   214‚Üí   214‚Üí  /** Simple checksum for migration verification */
   215‚Üí   215‚Üí  private simpleChecksum(sql: string): string {
   216‚Üí   216‚Üí    let hash = 0;
   217‚Üí   217‚Üí    const normalized = sql.replace(/\s+/g, ' ').trim();
   218‚Üí   218‚Üí    for (let i = 0; i < normalized.length; i++) {
   219‚Üí   219‚Üí      hash = ((hash << 5) - hash + normalized.charCodeAt(i)) | 0;
   220‚Üí   220‚Üí    }
   221‚Üí   221‚Üí    return Math.abs(hash).toString(16);
   222‚Üí   222‚Üí  }

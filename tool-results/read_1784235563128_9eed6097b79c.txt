   300→describe('Incremental Discovery', () => {
   301→  it('should discover new URLs on top of existing surface', async () => {
   302→    const mockFetcher = new MockFetcher();
   303→    mockFetcher.onUrl('https://example.com/page1', makeMockResponse(
   304→      '<html><body><a href="/page2">Page 2</a></body></html>'
   305→    ));
   306→    mockFetcher.onUrl('https://example.com/page2', makeMockResponse(
   307→      '<html><body><a href="/page3">Page 3</a></body></html>'
   308→    ));
   309→
   310→    const adapter = new KatanaAdapter({ httpClient: mockFetcher, scope: { ...DEFAULT_DISCOVERY_SCOPE, maxDepth: 5, maxUrls: 20 } });
   311→
   312→    // Phase 1: initial discovery
   313→    const surface1 = await adapter.incrementalDiscover(
   314→      createEmptyAttackSurface(),
   315→      ['https://example.com/page1'],
   316→      mockFetcher,
   317→    );
   318→    expect(surface1.urls.length).toBeGreaterThanOrEqual(1); // page2 found in page1 response
   319→
   320→    // Phase 2: incremental — discover from page2 (already processed, but crawlUrls
   321→    // doesn't re-fetch it since it's in the processed set from surface1).
   322→    // Instead, discover from a fresh URL not in the existing surface.
   323→    const surface2 = await adapter.incrementalDiscover(
   324→      surface1,
   325→      ['https://example.com/page2'],
   326→      mockFetcher,
   327→    );
   328→    // surface2 should have all of surface1 + anything new from page2's parse
   329→    // Since page2 IS in surface1's processed set, crawlUrls skips it.
   330→    // So surface2 === surface1 (no new URLs found).
   331→    expect(surface2.urls.length).toBe(surface1.urls.length);
   332→  });
